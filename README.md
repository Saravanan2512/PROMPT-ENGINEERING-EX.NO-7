# Python Prototype Code

```python
tasks = []

def add_task(task):
    tasks.append(task)
    print("Task Added Successfully!")

def show_tasks():
    print("\nToday's Tasks:")
    for i, task in enumerate(tasks, start=1):
        print(f"{i}. {task}")

while True:
    print("\n1. Add Task")
    print("2. Show Tasks")
    print("3. Exit")

    choice = input("Enter choice: ")

    if choice == '1':
        task = input("Enter task: ")
        add_task(task)

    elif choice == '2':
        show_tasks()

    elif choice == '3':
        print("Exiting Productivity Assistant")
        break

    else:
        print("Invalid Choice")
```
